    Create `backend/src/cache.py`:

    ```python
    import os
    import json
    from typing import Optional
    from supabase import create_client, Client

    TABLE = "laso_cache"

    def _get_client() -> Client:
        url = os.environ.get("SUPABASE_URL")
        key = os.environ.get("SUPABASE_KEY")
        if not url or not key:
            raise EnvironmentError("SUPABASE_URL and SUPABASE_KEY must be set")
        return create_client(url, key)

    def build_cache_key(ngay_sinh: str, gio_sinh: str, gender: str,
                        source: str, nam_xem: Optional[int] = None) -> str:
        """Build cache key per D-09:
        - cohoc: ngay_sinh_gio_sinh_gender (no year — data is static)
        - tuvivn: ngay_sinh_gio_sinh_gender_nam_xem (monthly data changes yearly)
        """
        key = f"{ngay_sinh}_{gio_sinh}_{gender}"
        if source == "tuvivn" and nam_xem is not None:
            key = f"{key}_{nam_xem}"
        return key

    def get_cached(cache_key: str, source: str) -> Optional[dict]:
        client = _get_client()
        result = (
            client.table(TABLE)
            .select("data")
            .eq("cache_key", cache_key)
            .eq("source", source)
            .limit(1)
            .execute()
        )
        if result.data:
            return result.data[0]["data"]
        return None

    def set_cached(cache_key: str, source: str, data: dict) -> None:
        client = _get_client()
        client.table(TABLE).upsert({
            "cache_key": cache_key,
            "source": source,
            "data": data,
        }).execute()
    ```

    The app uses a single shared Chrome driver initialized at startup.

    **Imports — do NOT import GIO_SINH from cohoc_scraper (it is an internal implementation detail).
    Import only what is needed:**

    ```python
    from contextlib import asynccontextmanager
    from datetime import datetime
    from typing import Optional
    import logging

    from fastapi import FastAPI, HTTPException
    from fastapi.middleware.cors import CORSMiddleware

    from src.schemas import LaSoTuViInput, LaSoTuVi, CungInfo, CungDetail
    from src.cache import get_cached, set_cached, build_cache_key
    from src.scoring import compute_scores, check_alerts
    from src.scrapers.cohoc_scraper import (
        get_page_detail as cohoc_scrape,
        create_driver,
        get_gio_sinh_option,
    )
    from src.scrapers.tuvivn_scraper import get_page_detail as tuvivn_scrape

    logger = logging.getLogger(__name__)
    driver = None

    @asynccontextmanager
    async def lifespan(app: FastAPI):
        global driver
        driver = create_driver()
        logger.info("Chrome driver initialized")
        yield
        driver.quit()
        logger.info("Chrome driver closed")

    app = FastAPI(lifespan=lifespan)
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["*"],  # Tighten in production
        allow_methods=["*"],
        allow_headers=["*"],
    )
    ```

    **Endpoint: POST /api/la-so**

    Request body: LaSoTuViInput
    Response: dict with keys:
    - `laso_cohoc`: raw cohoc scrape result (dict with cung + cung_10yrs)
    - `laso_tuvivn`: LaSoTuVi.model_dump() from tuvi.vn
    - `scores`: compute_scores() result
    - `alerts`: check_alerts() result
    - `cache_hit`: bool (True if both sources returned from cache)

    **Logic (implement in this exact order):**

    1. Parse dob from ngay_sinh ("dd/mm/yyyy") using `datetime.strptime(body.ngay_sinh, "%d/%m/%Y")`.

    2. Parse gio_sinh string to get the hour for `get_gio_sinh_option`. The format is "7h0" (hour + "h" + minutes).
       Extract the integer hour and build a datetime with that hour:
       ```python
       gio_hour = int(body.gio_sinh.split("h")[0])
       gio_dt = dob.replace(hour=gio_hour, minute=0, second=0, microsecond=0)
       gio_sinh_option = get_gio_sinh_option(gio_dt)
       ```

    3. Build cohoc cache key, check cache:
       ```python
       cohoc_key = build_cache_key(body.ngay_sinh, body.gio_sinh, body.gender, "cohoc")
       cohoc_raw = get_cached(cohoc_key, "cohoc")
       cohoc_cache_hit = cohoc_raw is not None
       ```

    4. If cache miss: scrape cohoc and cache result:
       ```python
       if not cohoc_cache_hit:
           cohoc_raw = cohoc_scrape(driver, dob, gio_sinh_option, body.gender)
           set_cached(cohoc_key, "cohoc", cohoc_raw)
       ```
       cohoc_raw is a dict: `{"cung": [list of dicts], "cung_10yrs": [list of dicts]}`

    5. Build tuvivn cache key (with nam_xem), check cache, scrape if miss:
       ```python
       tuvivn_key = build_cache_key(body.ngay_sinh, body.gio_sinh, body.gender, "tuvivn", body.nam_xem)
       tuvivn_raw = get_cached(tuvivn_key, "tuvivn")
       tuvivn_cache_hit = tuvivn_raw is not None
       if not tuvivn_cache_hit:
           tuvivn_result = tuvivn_scrape(driver, dob, body.gender, body.full_name, body.nam_xem)
           tuvivn_raw = tuvivn_result.model_dump()
           set_cached(tuvivn_key, "tuvivn", tuvivn_raw)
       else:
           tuvivn_result = LaSoTuVi(**tuvivn_raw)
       ```

    6. Convert cohoc raw dicts to CungDetail objects before assembling CungInfo.
       cohoc_scrape returns raw dicts, NOT CungDetail objects — Pydantic validation requires typed objects:
       ```python
       cung_chung = [CungDetail(**d) for d in cohoc_raw["cung"]]
       cung_10yrs = [CungDetail(**d) for d in cohoc_raw["cung_10yrs"]]
       cung_info = CungInfo(
           cung_chung=cung_chung,
           cung_10yrs=cung_10yrs,
           cung_12months=tuvivn_result.cung.cung_12months,
       )
       ```

    7. Compute scores and alerts:
       ```python
       scores = compute_scores(cung_info)
       alerts = check_alerts(cung_info)
       ```

    8. Return response dict:
       ```python
       return {
           "laso_cohoc": cohoc_raw,
           "laso_tuvivn": tuvivn_raw,
           "scores": scores,
           "alerts": alerts,
           "cache_hit": cohoc_cache_hit and tuvivn_cache_hit,
       }
       ```

    Error handling:
    - ValueError from scraper → HTTPException(422, detail=str(e))
    - RuntimeError from scraper → HTTPException(503, detail=str(e))
    - Catch-all → HTTPException(500, detail="Internal error")
